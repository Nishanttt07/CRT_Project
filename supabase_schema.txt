-- ==========================================
-- FULL IDEMPOTENT SUPABASE DATABASE SCHEMA
-- ==========================================

-- 1. Create the public.users table (Base Profile)
CREATE TABLE IF NOT EXISTS public.users (
  id uuid NOT NULL REFERENCES auth.users (id) ON DELETE CASCADE PRIMARY KEY,
  username text NOT NULL,
  email text NOT NULL UNIQUE,
  phone text NOT NULL UNIQUE,
  created_at timestamp with time zone NULL DEFAULT now()
);

-- Enable RLS for users
ALTER TABLE public.users ENABLE ROW LEVEL SECURITY;

-- Idempotent RLS Policies for users
DROP POLICY IF EXISTS "Allow users to read their own profile" ON public.users;
CREATE POLICY "Allow users to read their own profile" 
  ON public.users FOR SELECT USING (auth.role() = 'authenticated');

DROP POLICY IF EXISTS "Allow users to update their own profile" ON public.users;
CREATE POLICY "Allow users to update their own profile" 
  ON public.users FOR UPDATE USING (auth.uid() = id);

-- 2. Secure RPC function to retrieve email by phone (Bypasses RLS for login)
CREATE OR REPLACE FUNCTION public.get_email_by_phone(phone_query text)
RETURNS text
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  user_email text;
BEGIN
  SELECT email INTO user_email FROM public.users WHERE phone = phone_query LIMIT 1;
  RETURN user_email;
END;
$$;

-- 3. Trigger function to automatically sync user metadata on signup
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  -- Insert into public.users
  INSERT INTO public.users (id, username, email, phone)
  VALUES (
    new.id,
    COALESCE(new.raw_user_meta_data->>'username', ''),
    COALESCE(new.email, ''),
    COALESCE(new.raw_user_meta_data->>'phone', new.phone, '')
  );

  -- Insert into public.shops if registering as a tailor
  IF COALESCE((new.raw_user_meta_data->>'is_tailor')::boolean, false) = true THEN
    INSERT INTO public.shops (id, shop_name, bio, latitude, longitude, address)
    VALUES (
      new.id,
      COALESCE(new.raw_user_meta_data->>'shop_name', 'Atelier'),
      COALESCE(new.raw_user_meta_data->>'shop_bio', ''),
      COALESCE((new.raw_user_meta_data->>'shop_latitude')::float, 51.5113),
      COALESCE((new.raw_user_meta_data->>'shop_longitude')::float, -0.1402),
      COALESCE(new.raw_user_meta_data->>'shop_address', '')
    );
  END IF;

  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Attach trigger securely (Drops the old one if it exists to prevent duplicates)
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();

-- ==========================================
-- TAILOR SYSTEM TABLES
-- ==========================================

-- 4. Shops / Tailor Profiles Table
CREATE TABLE IF NOT EXISTS public.shops (
    id UUID REFERENCES public.users(id) ON DELETE CASCADE PRIMARY KEY,
    shop_name TEXT NOT NULL,
    bio TEXT,
    latitude FLOAT,
    longitude FLOAT,
    address TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE public.shops ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Allow anyone to read shops" ON public.shops;
CREATE POLICY "Allow anyone to read shops" 
  ON public.shops FOR SELECT USING (true);

DROP POLICY IF EXISTS "Allow users to manage their own shop" ON public.shops;
CREATE POLICY "Allow users to manage their own shop" 
  ON public.shops FOR ALL USING (auth.uid() = id);

-- 5. Customer Measurement Vault (Using JSONB for flexible formats)
CREATE TABLE IF NOT EXISTS public.measurements (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    customer_id UUID REFERENCES public.users(id) ON DELETE CASCADE,
    tailor_id UUID REFERENCES public.shops(id) ON DELETE CASCADE,
    garment_type TEXT NOT NULL,
    metrics JSONB NOT NULL,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE public.measurements ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Allow customers to view their measurements" ON public.measurements;
CREATE POLICY "Allow customers to view their measurements" 
  ON public.measurements FOR SELECT USING (auth.uid() = customer_id);

DROP POLICY IF EXISTS "Allow tailors to manage measurements" ON public.measurements;
CREATE POLICY "Allow tailors to manage measurements" 
  ON public.measurements FOR ALL USING (auth.uid() = tailor_id);

-- 6. Orders Table
CREATE TABLE IF NOT EXISTS public.orders (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    customer_id UUID REFERENCES public.users(id) ON DELETE CASCADE,
    tailor_id UUID REFERENCES public.shops(id) ON DELETE CASCADE,
    garment_type TEXT NOT NULL,
    status TEXT DEFAULT 'pending_measurements',
    total_price DECIMAL(10, 2),
    amount_paid DECIMAL(10, 2) DEFAULT 0,
    expected_delivery DATE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE public.orders ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Allow customers to view their orders" ON public.orders;
CREATE POLICY "Allow customers to view their orders" 
  ON public.orders FOR SELECT USING (auth.uid() = customer_id);

DROP POLICY IF EXISTS "Allow tailors to manage orders" ON public.orders;
CREATE POLICY "Allow tailors to manage orders" 
  ON public.orders FOR ALL USING (auth.uid() = tailor_id);

-- 7. Inventory Table
CREATE TABLE IF NOT EXISTS public.inventory (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    tailor_id UUID REFERENCES public.shops(id) ON DELETE CASCADE,
    item_name TEXT NOT NULL,
    quantity FLOAT NOT NULL DEFAULT 0,
    unit TEXT DEFAULT 'units',
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE public.inventory ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Allow tailors to manage their inventory" ON public.inventory;
CREATE POLICY "Allow tailors to manage their inventory" 
  ON public.inventory FOR ALL USING (auth.uid() = tailor_id);
